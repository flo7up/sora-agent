python -m venv .venv
.venv/Scripts/activate


pip install --upgrade -r requirements.txt

# Freeze current virtualenv to requirements.txt
pip freeze > requirements.txt

# View freeze without writing
pip freeze

# Save with timestamp
pip freeze > requirements-$(date +%Y%m%d).txt  2> NUL

# Compare currently installed vs requirements.txt
pip freeze | sort > current.txt
sort requirements.txt > locked.txt
fc /N current.txt locked.txt

# List only top-level packages (optional, needs pipdeptree)
pip install pipdeptree
pipdeptree --freeze --warn silence > requirements-top.txt
